cashe

Пакет представляет собой реализацию потокобезопасной хеш таблицы типа map[string]interface{} котрая и является in-memory хранилищем кеша.

cashe.New(
    hash func([]byte) uint32,   // хеш функция
    init_bucketscount int,      // первоначальное ко-во элементов таблицы
    ifreadthenlive bool,        // продлевать время жизни элемента при обращении к нему
    duration time.Duration,     // гарантированное время жизни элемента
    callback func(key *string, val *interface{}) // функция выполняемая при удалении элемента по причине истечения срока его жизни в кеше
) Cache

создаваемый объект реализует интерфейс:

Cache interface {                 
        GetBucketsStat() (countitem uint64, countbucket uint32, stat [][2]int) // получить информацию по структуре кеша(кол-во бакетов) и статистику распределения записей.
        Get(string) interface{}          // получить запись
        Set(string, interface{}) bool    // записать в кеш
        Del(string) (val interface{})    // удалить из кеша
        Inc(string, float64) interface{} // инкремент/декремент значения  
        Save(io.Writer) error            // сохранить в писателя
        SaveFile(string) error           // сохранить в файл 
        Load(io.Reader) error            // загрузить из читателя
        LoadFile(string) error           // загрузить из файла
        Len() I_counter                  // объект реализующий счётчик для возможного лимитирования кол-ва принимаемых записей
    }

объект счётчика реализует интерфейс:

I_counter interface {
        Get() uint64
        GetLimit() uint64
        SetLimit(v uint64)
        Check() bool
    }

реализацией данного интерфейса является пакет counter

hash

Пакет является коллекцией быстрых некриптографических хеш функций используемых в пакете cashe
